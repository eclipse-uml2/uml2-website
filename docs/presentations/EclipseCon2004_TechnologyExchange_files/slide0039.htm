Overriding ResourceImpl#detached(EObject)
public void detached(EObject eObject) {
    synchronized (detachedEObjectToIDMap) {
        detachedEObjectToIDMap.put(eObject, super.getID(eObject));
        for (Iterator children = eObject.eAllContents(); children.hasNext();) {
            EObject child = (EObject) children.next();
            detachedEObjectToIDMap.put(child, super.getID(child));
        }
    }
    super.detached(eObject);
}
